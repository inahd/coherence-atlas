  atlas doctor               Repair and validate graph

if cmd == "today":
    run("atlas_meaning_today.py")

elif cmd == "extract":
    run("extract_text_knowledge.py",sys.argv[2])

elif cmd == "ingest":
    run("ingest_dataset.py",sys.argv[2])

elif cmd == "search":
    run("query_graph.py",sys.argv[2])

elif cmd=="docs":
    run("generate_docs.py")

\n\nFILE:/opt/atlas/scripts/ingest_dataset.py\nimport yaml
import sys
import os

BASE="/opt/atlas"
DATASETS=os.path.join(BASE,"datasets")
GRAPH=os.path.join(BASE,"graph","graph.yaml")

dataset=sys.argv[1]

file=os.path.join(DATASETS,f"{dataset}.yaml")

if not os.path.exists(file):
    print("Dataset not found:",file)
    exit()

data=yaml.safe_load(open(file))

graph=yaml.safe_load(open(GRAPH))

nodes=graph.get("nodes",{})
edges=graph.get("edges",[])

def add_node(name,data):
    if name not in nodes:
        nodes[name]=data

def add_edge(a,rel,b):
    edge={"from":a,"relation":rel,"to":b}
    if edge not in edges:
        edges.append(edge)

# ingest nodes
for n in data.get("nodes",{}):
    add_node(n,data["nodes"][n])

# ingest edges
for e in data.get("edges",[]):
    add_edge(e["from"],e["relation"],e["to"])

graph["nodes"]=nodes
graph["edges"]=edges

with open(GRAPH,"w") as f:
    yaml.dump(graph,f)

print("Dataset ingested:",dataset)

nakshatra=[

]

with open(f"{BASE}/datasets/nakshatra.yaml","w") as f:
    yaml.dump(nakshatra,f)

now=datetime.utcnow()
jd=swe.julday(now.year,now.month,now.day)

sun=swe.calc_ut(jd,swe.SUN)[0][0]
moon=swe.calc_ut(jd,swe.MOON)[0][0]

tithi=int(((moon-sun)%360)/12)+1
nak=int(moon/(360/27))

data={
"date":now.strftime("%Y-%m-%d"),
"tithi":tithi,
"nakshatra":nakshatra[nak],
"sun":sun,
"moon":moon
}

file=f"{BASE}/panchanga/{data['date']}.yaml"

with open(file,"w") as f:
    yaml.dump(data,f)

print("Atlas bootstrap complete")
print("Today's Nakshatra:",data["nakshatra"])
print("Today's Tithi:",data["tithi"])
